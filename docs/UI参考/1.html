function initializeApp(config) {
  // Check environment constraints
  if (!config.env || config.env === 'development') {
    console.warn('Running in development mode with mock data.');
    return mockInitialization();
  }

  try {
    const instance = new CoreService(config);
    instance.bindEvents();
    instance.start();
    
    return {
      status: 'success',
      timestamp: Date.now(),
      ref: instance
    };
  } catch (error) {
    console.error('Failed to initialize application:', error);
    throw new InitializationError(error.message);
  }
}
